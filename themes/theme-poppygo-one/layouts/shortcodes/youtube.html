{{- /* A simplified shortcode to embed a YouTube video. It focuses on basic embedding without advanced styling or alignment. @param {string} id - The YouTube video ID. (Required) @param {string} [title="YouTube video"] - The title for accessibility. @param {string} [loading="eager"] - The loading attribute of the iframe. */ -}} {{- $pc := .Page.Site.Config.Privacy.YouTube }} {{- $remoteErrID := "err-youtube-remote" }} {{- $referrerpolicy := "strict-origin-when-cross-origin" }} {{- if not $pc.Disable }} {{- with $id := or (.Get "id") (.Get 0) }} {{- /* Set defaults. */}} {{- $title := or (.Get "title") "YouTube video" }} {{- $loading := or (.Get "loading") "eager" }} {{- $allowFullScreen := true }} {{- $controls := 1 }} {{- $playsinline := 1 }} {{- /* Define src attribute and query parameters. */}} {{- $host := cond $pc.PrivacyEnhanced "www.youtube-nocookie.com" "www.youtube.com" }} {{- $src := printf "https://%s/embed/%s" $host $id }} {{- $queryParams := slice }} {{- if eq $controls 0 }} {{ $queryParams = $queryParams | append "controls=0" }} {{- end }} {{ $queryParams = $queryParams | append (printf "playsinline=%d" $playsinline) }} {{- if gt (len $queryParams) 0 }} {{- $src = printf "%s?%s" $src (delimit $queryParams "&") }} {{- end }} {{- /* Render the basic iframe. No custom CSS classes for sizing here. */ -}} {{- else }} {{- errorf "The %q shortcode requires an id argument. See %s" .Name .Position }} {{- end }} {{- end }}
